$ USERSPACE
/Users/$USER/                       # Home User directory
/Users/$USER/Programs               # User Programs/Executable directory
/Users/$USER/.local                 # User Settings and cache files

$ SYSTEM
/System/Rhoudveine                  # System Folder

/System/Rhoudveine/Booter           # Boot Folder (contains init executable and kernel)
/System/Rhoudveine/Booter/EFI       # EFI mounted Folder

/System/Rhoudveine/Programs         # System Programs

/System/Rhoudveine/Runtime          # Runtime Specific folder (created by kernel and dameon, meaning it is not stored on disk)
/System/Rhoudveine/Runtime/Device   # Devices
/System/Rhoudveine/Runtime/Process  # Running processes

/System/Rhoudveine/Library          # Libraries

/System/Rhoudveine/Databases        # System Database of system settings, configuration and installed programs
